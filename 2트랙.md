```tsx
function countryFlag(code?: string): string {
  if (!code || code.length !== 2) return '🌐'
  return [...code.toUpperCase()]
    .map((c) => String.fromCodePoint(c.charCodeAt(0) + 127397))
    .join('')
}

function formatDubai(value?: string | number | null) {
  if (!value) return '—'
  const date = typeof value === 'number' ? new Date(value * 1000) : new Date(value)
  return date.toLocaleString('en-AE', {
    timeZone: 'Asia/Dubai',
    month: 'short',
    day: 'numeric',
    hour: '2-digit',
    minute: '2-digit',
  })
}

function computeAnimTime(epochStart: number, elapsedMs: number) {
  const elapsedSec = (elapsedMs / 1000) % TRIPS_TIME_WINDOW_SECS
  return epochStart + elapsedSec
}

function buildPoiIntent(poi: PoiLocation): NavigationIntent {
  if (poi.category === 'HVDC_SITE' && poi.siteCode) {
    return {
      destinationId: 'map-site',
      page: 'sites',
      params: { site: poi.siteCode, tab: 'summary' },
    }
  }

  if (poi.category === 'PORT' || poi.category === 'AIRPORT') {
    return {
      destinationId: 'map-port',
      page: 'chain',
      params: { focus: 'port' },
    }
  }

  if (poi.category === 'WAREHOUSE') {
    return {
      destinationId: 'map-warehouse',
      page: 'chain',
      params: { focus: 'warehouse' },
    }
  }

  return {
    destinationId: 'map-mosb',
    page: 'chain',
    params: { focus: 'mosb' },
  }
}

function buildNodeIntent(node: NetworkNode | UaeNode): NavigationIntent {
  const siteCode = 'siteCode' in node ? node.siteCode : undefined

  if (siteCode) {
    return {
      destinationId: 'map-site',
      page: 'sites',
      params: { site: siteCode, tab: 'summary' },
    }
  }

  if (node.type === 'port' || node.type === 'airport') {
    return {
      destinationId: 'map-port',
      page: 'chain',
      params: { focus: 'port' },
    }
  }

  if (node.type === 'warehouse') {
    return {
      destinationId: 'map-wh',
      page: 'chain',
      params: { focus: 'warehouse' },
    }
  }

  if (node.type === 'customs') {
    return {
      destinationId: 'map-customs',
      page: 'pipeline',
      params: { focus: 'customs' },
    }
  }

  return {
    destinationId: 'map-mosb',
    page: 'chain',
    params: { focus: 'mosb' },
  }
}

function createTripsLayer(
  trips: TripData[],
  currentTime: number,
  visible: boolean,
  highlightedTripId?: string | null,
) {
  return new TripsLayer<TripData>({
    id: 'network-trips',
    data: trips,
    visible,
    pickable: true,
    getPath: (d) => d.path,
    getTimestamps: (d) =>
      d.timestamps && d.timestamps.length === d.path.length
        ? d.timestamps
        : d.path.map((_, i) => i * 18),
    getColor: (d) => {
      if (highlightedTripId && d.id === highlightedTripId) return [255, 255, 255, 240]
      if (d.highlighted) return [255, 255, 255, 220]
      if ((d.flowCode ?? 0) >= 3) return [246, 180, 69, 185]
      return [83, 191, 255, 165]
    },
    opacity: 0.92,
    widthMinPixels: 2,
    rounded: true,
    trailLength: 120,
    currentTime,
    parameters: {
      depthTest: false,
      blend: true,
      blendFunc: [770, 771],
    },
    updateTriggers: {
      getColor: [highlightedTripId],
    },
  })
}

function createLabelLayer(
  globalNodes: NetworkNode[],
  uaeNodes: UaeNode[],
  pois: PoiLocation[],
  track: MapTrack,
) {
  const globalData = globalNodes
    .filter((n) => n.type === 'site' || n.type === 'port' || n.type === 'airport' || n.type === 'mosb')
    .map((n) => ({
      id: `g-${n.id}`,
      text: n.siteCode || n.name,
      lon: n.lon,
      lat: n.lat,
      strong: n.type === 'site',
    }))

  const uaeData = uaeNodes
    .filter((n) => n.type === 'site' || n.type === 'port' || n.type === 'customs' || n.type === 'mosb')
    .map((n) => ({
      id: `u-${n.id}`,
      text: ('siteCode' in n && n.siteCode) ? n.siteCode : n.name,
      lon: n.lon,
      lat: n.lat,
      strong: n.type === 'site',
    }))

  const poiData = pois
    .filter((p) => p.category === 'HVDC_SITE' || p.category === 'PORT')
    .map((p) => ({
      id: `p-${p.id}`,
      text: p.siteCode || p.name,
      lon: p.lon,
      lat: p.lat,
      strong: p.category === 'HVDC_SITE',
    }))

  const data = track === 'global' ? [...globalData, ...poiData] : [...uaeData]

  return new TextLayer<(typeof data)[number]>({
    id: 'network-labels',
    data,
    visible: true,
    pickable: false,
    getPosition: (d) => [d.lon, d.lat],
    getText: (d) => d.text,
    getSize: (d) => (d.strong ? 16 : 13),
    sizeUnits: 'pixels',
    getColor: [244, 247, 255, 235],
    getTextAnchor: 'middle',
    getAlignmentBaseline: 'bottom',
    getPixelOffset: [0, -18],
    fontWeight: 700,
    background: true,
    getBackgroundColor: [11, 19, 36, 170],
    getBorderColor: [255, 255, 255, 24],
    getBorderWidth: 1,
    backgroundPadding: [8, 4],
    parameters: {
      depthTest: false,
    },
  })
}

/* =========================
 * 메인
 * ========================= */

interface OverviewMapProps {
  globalNodes?: NetworkNode[]
  globalEdges?: NetworkEdge[]
  uaeNodes?: UaeNode[]
  uaeEdges?: UaeRouteEdge[]
  pois?: PoiLocation[]
  onNavigateIntent?: (intent: NavigationIntent) => void
}

export function OverviewMap({
  globalNodes = [],
  globalEdges = [],
  uaeNodes = [],
  uaeEdges = [],
  pois = POI_LOCATIONS,
  onNavigateIntent,
}: OverviewMapProps) {
  const mapContainerRef = useRef<HTMLDivElement>(null)
  const mapRef = useRef<maplibregl.Map | null>(null)
  const overlayRef = useRef<MapboxOverlay | null>(null)
  const didFitBoundsRef = useRef(false)
  const animFrameRef = useRef<number | null>(null)
  const animStartRef = useRef<number>(Date.now())

  const [zoom, setZoom] = useState(INITIAL_VIEW.zoom)
  const [mapTrack, setMapTrack] = useState<MapTrack>('uae-ops')
  const [toggles, setToggles] = useState<MapLayerToggles>({
    showOriginArcs: true,
    showTrips: true,
    showCustoms: true,
    showWarehouses: true,
    showMosb: true,
    showSites: true,
  })
  const [tooltip, setTooltip] = useState<TooltipState | null>(null)
  const [selectedNodeId, setSelectedNodeId] = useState<string | null>(null)
  const [highlightedTripId, setHighlightedTripId] = useState<string | null>(null)
  const [tripsData, setTripsData] = useState<TripData[]>([])
  const [originsData, setOriginsData] = useState<OriginEntry[]>([])
  const [animTime, setAnimTime] = useState(ANIM_EPOCH_START)

  const effectiveGlobalNodes = useMemo(
    () => (globalNodes.length > 0 ? globalNodes : FALLBACK_GLOBAL_NODES),
    [globalNodes],
  )
  const effectiveGlobalEdges = useMemo(
    () => (globalEdges.length > 0 ? globalEdges : FALLBACK_GLOBAL_EDGES),
    [globalEdges],
  )
  const effectiveUaeNodes = useMemo(
    () => (uaeNodes.length > 0 ? uaeNodes : FALLBACK_UAE_NODES),
    [uaeNodes],
  )
  const effectiveUaeEdges = useMemo(
    () => (uaeEdges.length > 0 ? uaeEdges : FALLBACK_UAE_EDGES),
    [uaeEdges],
  )

  const navigate = useCallback(
    (intent: NavigationIntent) => {
      onNavigateIntent?.(intent)
    },
    [onNavigateIntent],
  )

  const handleToggleChange = useCallback(
    <K extends keyof MapLayerToggles>(key: K, value: MapLayerToggles[K]) => {
      setToggles((prev) => ({ ...prev, [key]: value }))
    },
    [],
  )

  const handleHover = useCallback((info: PickingInfo) => {
    if (!info?.object) {
      setTooltip(null)
      return
    }

    const layerId = info.layer?.id ?? ''

    if (layerId === 'network-arcs' || layerId === 'uae-route-layer') {
      const arc = info.object as any
      setTooltip({
        kind: 'arc',
        x: info.x,
        y: info.y,
        arc: {
          country: arc.country,
          count: arc.count,
          flowCode: arc.flowCode,
          volume: arc.volume,
          routeType: arc.routeType,
        },
      })
      return
    }

    if (layerId === 'network-trips') {
      setTooltip({
        kind: 'trip',
        x: info.x,
        y: info.y,
        trip: info.object as TripData,
      })
      return
    }

    if (
      layerId === 'network-nodes' ||
      layerId === 'customs-node-layer'
    ) {
      setTooltip({
        kind: 'node',
        x: info.x,
        y: info.y,
        node: info.object as NetworkNode | UaeNode,
      })
      return
    }

    setTooltip(null)
  }, [])

  const handleNodeClick = useCallback(
    (info: PickingInfo) => {
      const node = info.object as NetworkNode | UaeNode | undefined
      if (!node) return
      setSelectedNodeId(node.id)
      navigate(buildNodeIntent(node))
    },
    [navigate],
  )

  const handleTrackChange = useCallback((track: MapTrack) => {
    setMapTrack(track)
    setSelectedNodeId(null)
    setTooltip(null)
    didFitBoundsRef.current = false

    if (mapRef.current) {
      const bounds = track === 'global' ? GLOBAL_BOUNDS : UAE_BOUNDS
      mapRef.current.fitBounds(bounds, {
        padding: 52,
        duration: 500,
      })
    }
  }, [])

  useEffect(() => {
    if (!mapContainerRef.current || mapRef.current) return

    const map = new maplibregl.Map({
      container: mapContainerRef.current,
      style: MAP_STYLE,
      center: [INITIAL_VIEW.longitude, INITIAL_VIEW.latitude],
      zoom: INITIAL_VIEW.zoom,
      attributionControl: false,
      pitch: 0,
      bearing: 0,
      antialias: true,
    })

    map.addControl(
      new maplibregl.NavigationControl({
        showCompass: false,
        visualizePitch: false,
      }),
      'bottom-right',
    )

    map.on('zoomend', () => setZoom(map.getZoom()))
    map.on('moveend', () => setZoom(map.getZoom()))

    map.once('load', () => {
      if (!didFitBoundsRef.current) {
        map.fitBounds(UAE_BOUNDS, { padding: 48, duration: 0 })
        didFitBoundsRef.current = true
        setZoom(map.getZoom())
      }
    })

    const overlay = new MapboxOverlay({
      interleaved: false,
      layers: [],
    })

    map.addControl(overlay as unknown as maplibregl.IControl)

    mapRef.current = map
    overlayRef.current = overlay

    return () => {
      map.remove()
      mapRef.current = null
      overlayRef.current = null
    }
  }, [])

  useEffect(() => {
    fetch('/api/shipments/trips')
      .then((r) => r.json())
      .then((j) => setTripsData(j.trips ?? []))
      .catch(() => {
        setTripsData([
          {
            id: 'trip-1',
            vendor: 'Hitachi',
            flowCode: 4,
            etaUnix: Math.floor(Date.now() / 1000) + 3600 * 24,
            highlighted: true,
            path: [
              [54.89, 24.81],
              [54.58, 24.36],
              [54.53, 24.31],
              [54.94, 24.53],
            ],
          },
          {
            id: 'trip-2',
            vendor: 'Prysmian',
            flowCode: 1,
            etaUnix: Math.floor(Date.now() / 1000) + 3600 * 36,
            path: [
              [54.89, 24.81],
              [54.96, 24.98],
            ],
          },
        ])
      })
  }, [])

  useEffect(() => {
    fetch('/api/chain/summary')
      .then((r) => r.json())
      .then((j) => setOriginsData(j.origins ?? []))
      .catch(() => {
        setOriginsData([
          {
            id: 'origin-1',
            country: 'SE',
            count: 430,
            sourceLon: 18.07,
            sourceLat: 59.33,
            targetLon: 54.89,
            targetLat: 24.81,
          },
          {
            id: 'origin-2',
            country: 'DE',
            count: 265,
            sourceLon: 8.68,
            sourceLat: 50.11,
            targetLon: 54.89,
            targetLat: 24.81,
          },
        ])
      })
  }, [])

  useEffect(() => {
    if (tripsData.length === 0) return

    animStartRef.current = Date.now()

    const tick = () => {
      const elapsed = Date.now() - animStartRef.current
      setAnimTime(computeAnimTime(ANIM_EPOCH_START, elapsed))
      animFrameRef.current = requestAnimationFrame(tick)
    }

    animFrameRef.current = requestAnimationFrame(tick)

    return () => {
      if (animFrameRef.current !== null) {
        cancelAnimationFrame(animFrameRef.current)
        animFrameRef.current = null
      }
    }
  }, [tripsData.length])

  useEffect(() => {
    if (!overlayRef.current) return

    const globalLayers = [
      toggles.showOriginArcs
        ? createArcLayer(effectiveGlobalEdges, effectiveGlobalNodes, {
            visible: true,
            onHover: handleHover,
          })
        : null,

      ...createNodeLayer(
        effectiveGlobalNodes.filter((node) => {
          if (!toggles.showMosb && node.type === 'mosb') return false
          if (!toggles.showSites && node.type === 'site') return false
          if (!toggles.showWarehouses && node.type === 'warehouse') return false
          return true
        }),
        {
          visible: true,
          showLabels: false,
          selectedNodeId,
          onHover: handleHover,
          onClick: handleNodeClick,
        },
      ),

      ...createStatusRingLayer(
        effectiveGlobalNodes.filter((node) => {
          if (!toggles.showMosb && node.type === 'mosb') return false
          if (!toggles.showSites && node.type === 'site') return false
          return true
        }),
        {
          visible: true,
          selectedNodeId,
          pulse: true,
        },
      ),

      toggles.showTrips
        ? createTripsLayer(tripsData, animTime, true, highlightedTripId)
        : null,

      createLabelLayer(effectiveGlobalNodes, effectiveUaeNodes, pois, 'global'),
    ].filter(Boolean)

    const uaeLayers = [
      createUaeRouteLayer(effectiveUaeEdges, effectiveUaeNodes, {
        visible: true,
        showCustoms: toggles.showCustoms,
        showWarehouses: toggles.showWarehouses,
        showMosb: toggles.showMosb,
        showSites: toggles.showSites,
        onHover: handleHover,
      }),

      ...createCustomsNodeLayer(effectiveUaeNodes, {
        visible: toggles.showCustoms,
        selectedNodeId,
        showLabels: true,
        onHover: handleHover,
        onClick: handleNodeClick,
      }),

      ...createNodeLayer(
        effectiveUaeNodes
          .filter((node) => node.type !== 'customs')
          .filter((node) => {
            if (!toggles.showWarehouses && node.type === 'warehouse') return false
            if (!toggles.showMosb && node.type === 'mosb') return false
            if (!toggles.showSites && node.type === 'site') return false
            return true
          }) as any,
        {
          visible: true,
          showLabels: false,
          selectedNodeId,
          onHover: handleHover,
          onClick: handleNodeClick,
        },
      ),

      ...createStatusRingLayer(
        effectiveUaeNodes
          .filter((node) => node.type !== 'customs')
          .filter((node) => {
            if (!toggles.showMosb && node.type === 'mosb') return false
            if (!toggles.showSites && node.type === 'site') return false
            return true
          }) as any,
        {
          visible: true,
          selectedNodeId,
          pulse: true,
        },
      ),

      toggles.showTrips
        ? createTripsLayer(tripsData, animTime, true, highlightedTripId)
        : null,

      createLabelLayer(effectiveGlobalNodes, effectiveUaeNodes, pois, 'uae-ops'),
    ].filter(Boolean)

    overlayRef.current.setProps({
      layers: mapTrack === 'global' ? globalLayers : uaeLayers,
      onHover: handleHover,
    })
  }, [
    mapTrack,
    toggles,
    effectiveGlobalEdges,
    effectiveGlobalNodes,
    effectiveUaeEdges,
    effectiveUaeNodes,
    pois,
    tripsData,
    animTime,
    selectedNodeId,
    highlightedTripId,
    handleHover,
    handleNodeClick,
  ])

  const stats = useMemo(() => {
    if (mapTrack === 'global') {
      const ports = effectiveGlobalNodes.filter((x) => x.type === 'port' || x.type === 'airport').length
      const sites = effectiveGlobalNodes.filter((x) => x.type === 'site').length

      return {
        origins: originsData.length,
        originCases: originsData.reduce((sum, x) => sum + x.count, 0),
        ports,
        sites,
        trips: tripsData.length,
      }
    }

    const ports = effectiveUaeNodes.filter((x) => x.type === 'port' || x.type === 'airport').length
    const customs = effectiveUaeNodes.filter((x) => x.type === 'customs').length
    const warehouses = effectiveUaeNodes.filter((x) => x.type === 'warehouse').length
    const mosb = effectiveUaeNodes.filter((x) => x.type === 'mosb').length
    const sites = effectiveUaeNodes.filter((x) => x.type === 'site').length

    return {
      ports,
      customs,
      warehouses,
      mosb,
      sites,
      trips: tripsData.length,
    }
  }, [mapTrack, effectiveGlobalNodes, effectiveUaeNodes, originsData, tripsData.length])

  return (
    <div className="relative h-full w-full overflow-hidden rounded-[20px]">
      <div
        ref={mapContainerRef}
        className="h-full w-full"
        aria-label="Overview network map"
        tabIndex={-1}
      />

      <MapTrackSwitch
        track={mapTrack}
        onTrackChange={handleTrackChange}
        toggles={toggles}
        onToggleChange={handleToggleChange}
      />

      {/* 좌측 상단 요약 패널 */}
      <div className="pointer-events-none absolute left-4 top-[146px] z-20 hidden w-[220px] rounded-[20px] border border-white/10 bg-black/36 p-4 backdrop-blur-md xl:block">
        <div className="mb-4 flex items-center gap-2">
          <span className="text-base text-white/90">◉</span>
          <div>
            <div className="text-sm font-semibold text-white">
              {mapTrack === 'global' ? 'Global Network' : 'UAE Ops Network'}
            </div>
            <div className="text-[11px] text-white/45">
              {mapTrack === 'global' ? 'Origin → POD → Site' : 'Port → Customs → WH → MOSB → Site'}
            </div>
          </div>
        </div>

        <div className="space-y-3">
          <LegendDot colorClass="bg-cyan-400" label="Port / Airport" />
          {mapTrack === 'uae-ops' ? <LegendDot colorClass="bg-sky-300" label="Customs" /> : null}
          <LegendDot colorClass="bg-amber-400" label="Warehouse" />
          <LegendDot colorClass="bg-orange-400" label="MOSB" />
          <LegendDot colorClass="bg-emerald-400" label="HVDC Site" />
        </div>

        <div className="my-4 h-px bg-white/10" />

        <div className="grid grid-cols-2 gap-2">
          {mapTrack === 'global' ? (
            <>
              <MetricChip label="Origins" value={String((stats as any).origins ?? 0)} />
              <MetricChip label="Cases" value={((stats as any).originCases ?? 0).toLocaleString()} />
              <MetricChip label="Ports" value={String((stats as any).ports ?? 0)} />
              <MetricChip label="Sites" value={String((stats as any).sites ?? 0)} />
              <MetricChip label="Trips" value={String((stats as any).trips ?? 0)} />
              <MetricChip label="Mode" value="Global" />
            </>
          ) : (
            <>
              <MetricChip label="Ports" value={String((stats as any).ports ?? 0)} />
              <MetricChip label="Customs" value={String((stats as any).customs ?? 0)} />
              <MetricChip label="WH" value={String((stats as any).warehouses ?? 0)} />
              <MetricChip label="MOSB" value={String((stats as any).mosb ?? 0)} />
              <MetricChip label="Sites" value={String((stats as any).sites ?? 0)} />
              <MetricChip label="Trips" value={String((stats as any).trips ?? 0)} />
            </>
          )}
        </div>
      </div>

      {/* 우하단 상태 */}
      <div className="pointer-events-none absolute bottom-4 right-4 z-20 rounded-2xl border border-white/10 bg-black/38 px-3 py-2 backdrop-blur-md">
        <div className="flex items-center gap-2">
          <span className="h-2.5 w-2.5 rounded-full bg-emerald-400" />
          <div className="text-xs text-white/78">
            {mapTrack === 'global' ? 'Global Supply View' : 'UAE Ops View'}
          </div>
        </div>
      </div>

      {tooltip ? <TooltipPanel tooltip={tooltip} /> : null}
    </div>
  )
}

export default OverviewMap
```

## 핵심

이 통합본은 한 파일 안에 아래를 모두 포함합니다.

* `Global / UAE Ops` 스위치
* `Origin Arc / Trips / Customs / WH / MOSB / Sites` on/off
* `TripsLayer`
* `LabelLayer`
* `Tooltip UI`
* `Global Track / UAE Ops Track` 레이어 분기

## 파일 의존성

이 파일이 전제하는 것은 딱 5개입니다.

* `createArcLayer.ts`
* `createNodeLayer.ts`
* `createStatusRingLayer.ts`
* `createUaeRouteLayer.ts`
* `createCustomsNodeLayer.ts`
